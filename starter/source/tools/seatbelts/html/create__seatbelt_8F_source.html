166  mid = ixr(5,i)
167  IF (mid > 0) THEN
168  mtyp = ipm(2,mid)

226  ENDIF
227  ENDDO

229  CALL ancmsg(msgid=2083,
230  . msgtype=msgerror,
231  . anmode=aninfo_blind_1,
232  . i1=igeo(1,i),i2=iskn(liskn*(isk-1)+4))
233  ENDIF
234  ENDIF
235  ENDIF
236  ENDIF
237  ENDDO
238 C
239  DEALLOCATE(tag_prop_2d)
240 C
241 C----------------------------------------------------------------------------
242 C--- Loop on elements on edges of seatbelt
243 C----------------------------------------------------------------------------
244 C
245 C-- if nshell = 1 and nspring = 1 -> node in corner of 2D belt
246 C-- if nshell = 2 and nspring = 1 -> node on edge of 2D belt
247 C-- if nshell = 0 and nspring = 1 -> node at end of 1D belt
248 C
249  my_allocate(tag_nod,numnod)
250  tag_nod(1:numnod) = 0
251  compt_belt_end = 0
252  compt_fram = 0
253  DO i=1,numnod
254  IF ((tag_nod_shell(i) < 2).AND.(tag_nod_spring(i)==1).AND.(tag_nod(i)==0)) THEN
255  compt_belt_end = compt_belt_end + 1
256  compt_fram = compt_fram + 1
257  tag_nod(i) = 1
258  IF (tag_nod_shell(i) == 1) THEN
259  next_node = i
260  DO WHILE(next_node > 0)
261  node_cur = next_node
262  next_node = 0
263  DO k=knod2elc(node_cur)+1,knod2elc(node_cur+1)
264  elem_cur = nod2elc(k)
265  mid = ixc(1,elem_cur)
266  mtyp = ipm(2,mid)
267  IF (mtyp==119) THEN
268  DO j=2,5
269  IF ((tag_nod_spring(ixc(j,elem_cur))==1).AND.(tag_nod(ixc(j,elem_cur))==0)) THEN
270 C-- next node on transverse edge of seatbelt
271  next_node = ixc(j,elem_cur)
272  tag_nod(next_node) = 1
273  compt_fram = compt_fram + 1
274  ENDIF
275  ENDDO
276  ENDIF
277  ENDDO
278  ENDDO
279  ENDIF
280  ENDIF
281  ENDDO
282 C
283  tag_nod(1:numnod) = 0
284  my_allocate(belt_end_nfram,compt_belt_end)
285  my_allocate(belt_end_addr,compt_belt_end)
286  my_allocate(fram_tab,compt_fram)
287  my_allocate(belt_end_section,compt_belt_end)
288  belt_end_nfram(1:compt_belt_end) = 0
289  belt_end_addr(1:compt_belt_end) = 0
290  belt_end_section(1:compt_belt_end) = zero
291  fram_tab(1:compt_fram) = 0
292  compt_belt_end = 0
293  compt_fram = 0
294  DO i=1,numnod
295  IF ((tag_nod_shell(i) < 2).AND.(tag_nod_spring(i)==1).AND.(tag_nod(i)==0)) THEN
296  compt_belt_end = compt_belt_end + 1
297  compt_fram = compt_fram + 1
298  tag_nod(i) = 1
299  belt_end_nfram(compt_belt_end) = 1
300  belt_end_addr(compt_belt_end) = compt_fram
301  fram_tab(compt_fram) = i
302  IF (tag_nod_shell(i) == 1) THEN
303 C
304 C-- determination of longitudinal direction using spring connected to corner of seatblet
305  DO k=knod2el1d(i)+1,knod2el1d(i+1)
306  IF (nod2el1d(k) > numelt+numelp) THEN
307  elem_cur = nod2el1d(k)-numelt-numelp
308  mid = ixr(5,elem_cur)
309  IF (mid > 0) THEN
310  mtyp = ipm(2,mid)
311  IF ((mtyp == 114).AND.(ixr(2,elem_cur)/= i)) THEN
312  node_longi = ixr(2,elem_cur)
313  ELSEIF (mtyp == 114) THEN
314  node_longi = ixr(3,elem_cur)
315  ENDIF
316  ENDIF
317  ENDIF
318  ENDDO
319  dist2 = (x(1,i)-x(1,node_longi))**2+(x(2,i)-x(2,node_longi))**2+(x(3,i)-x(3,node_longi))**2
320  longi_direction(1) = (x(1,i)-x(1,node_longi))/sqrt(max(em20,dist2))
321  longi_direction(2) = (x(2,i)-x(2,node_longi))/sqrt(max(em20,dist2))
322  longi_direction(3) = (x(3,i)-x(3,node_longi))/sqrt(max(em20,dist2))
323 C
324  next_node = i
325  DO WHILE(next_node > 0)
326  node_cur = next_node
327  next_node = 0
328  DO k=knod2elc(node_cur)+1,knod2elc(node_cur+1)
329  elem_cur = nod2elc(k)
330  mid = ixc(1,elem_cur)
331  mtyp = ipm(2,mid)
332  IF (mtyp==119) THEN
333  DO j=2,5
334  IF ((tag_nod_spring(ixc(j,elem_cur))==1).AND.(tag_nod(ixc(j,elem_cur))==0)) THEN
335 C-- next node on transverse edge of seatbelt
336  next_node = ixc(j,elem_cur)
337  tag_nod(next_node) = 1
338  compt_fram = compt_fram + 1
339  fram_tab(compt_fram) = next_node
340  ENDIF
341  ENDDO
342  ENDIF
343  ENDDO
344  IF (next_node > 0) THEN
345 C-- seatbelt section is incremented
346  dist2 = (x(1,node_cur)-x(1,next_node))**2+(x(2,node_cur)-x(2,next_node))**2
347  . +(x(3,node_cur)-x(3,next_node))**2
348  edge_direction(1) = (x(1,node_cur)-x(1,next_node))/sqrt(max(em20,dist2))
349  edge_direction(2) = (x(2,node_cur)-x(2,next_node))/sqrt(max(em20,dist2))
350  edge_direction(3) = (x(3,node_cur)-x(3,next_node))/sqrt(max(em20,dist2))
351  scal = longi_direction(1)*edge_direction(1)+longi_direction(2)*edge_direction(2)
352  . +longi_direction(3)*edge_direction(3)
353  dist2 = dist2*(one-scal*scal)
354  ipid = ixc(6,elem_cur)
355  belt_end_section(compt_belt_end) = belt_end_section(compt_belt_end) + sqrt(max(em20,dist2))*geo(1,ipid)
356  ENDIF
357  ENDDO
358  belt_end_nfram(compt_belt_end) = compt_fram - belt_end_addr(compt_belt_end) + 1
359  ENDIF
360  ENDIF
361  ENDDO
362 C
363 C DO I=1,COMPT_BELT_END
364 C DO J=1,BELT_END_NFRAM(I)
365 C print *,"-->",I,ITAB(FRAM_TAB(BELT_END_ADDR(I)+J-1))
366 C ENDDO
367 C print *,"SECTION",BELT_END_SECTION(I)
368 C ENDDO
369 C
370  DEALLOCATE(tag_nod_spring,tag_nod_shell)
371 C
372  my_allocate(tag_res,numelr)
373  my_allocate(tag_fram_seatbelt,compt_belt_end)
374  my_allocate(nnod_fram_seatbelt,compt_belt_end)
375  tag_nod(1:numnod) = 0
376  tag_res(1:numelr) = 0
377  seatbelt_id = 0
378  flag = 0
379  nb_2d_seatbelt = 0
380  tag_fram_seatbelt(1:compt_belt_end) = 0
381  nnod_fram_seatbelt(1:compt_belt_end) = 0
382 C
383 C----------------------------------------------------------------------------
384 C--- Loop on seatblet elements in longitudinal direction
385 C----------------------------------------------------------------------------
386 C
387  IF (compt_belt_end == 0) THEN
388  CALL ancmsg(msgid=2099,
389  . msgtype=msgerror,
390  . anmode=aninfo_blind_1)

394 C
395  DO i=1,compt_belt_end
396 C

398 C

402 C

404 C
405  DO j=1,belt_end_nfram(i)
406 C
407  nnod = nnod + 1

409  ndir = 0
410 C
411  DO k=knod2el1d(nod_start)+1,knod2el1d(nod_start+1)
412  IF (nod2el1d(k) > numelt+numelp) THEN
413  elem_cur = nod2el1d(k)-numelt-numelp
414  mid = ixr(5,elem_cur)
415  IF (mid > 0) THEN
416  mtyp = ipm(2,mid)
417  IF (mtyp == 114) THEN

423  2 nod_start, elem_cur, tag_res, tag_nod,
424  3 seatbelt_id,flag, nnod, ipm,
425  4 nb_elem_1d, nb_branch, branch_tab, branch_cpt,
426  5 numelt, numelp, numelr, nummat,
427  6 sixx, numnod, nretractor, nslipring,

429 
430 C-- Loop on subranch (only if no sliprings and no retractors)
431  DO WHILE(nb_branch > 0)
432  nod_start = branch_tab(2*(branch_cpt-nb_branch)+1)
433  elem_cur = branch_tab(2*(branch_cpt-nb_branch)+2)

436  1 ixr, itab, knod2el1d, nod2el1d,
437  2 nod_start, elem_cur, tag_res, tag_nod,
438  3 seatbelt_id,flag, nnod, ipm,
439  4 nb_elem_1d, nb_branch, branch_tab, branch_cpt,
440  5 numelt, numelp, numelr, nummat,
441  6 sixx, numnod, nretractor, nslipring,
442  7 npropmi)
443  ENDDO
444 C
445  ENDIF
446  ENDIF
447  ENDIF
448  ENDDO
449 C
450  ENDDO
451 C
452  tag_fram_seatbelt(i) = seatbelt_id
453  nnod_fram_seatbelt(i) = nnod
454 C
455  ELSEIF(belt_end_nfram(i) > 1) THEN
456 C-- check of frames (2D sliprings)
457  compt = 0
458  DO j=1,belt_end_nfram(i)
459  IF (tag_nod(fram_tab(belt_end_addr(i))) /= 0) compt = compt + 1
460  ENDDO
461  IF (compt /= belt_end_nfram(i)) THEN
462  CALL ancmsg(msgid=2073,
463  . msgtype=msgerror,
464  . anmode=aninfo_blind_1,
465  . i1=slipring(i)%ID)
466  ENDIF
467 C
468  ENDIF
469 C
470  ENDDO
471 C
472  DEALLOCATE(branch_tab)
473 C
474 C----------------------------------------------------------------------------
475 C--- Filling of seatbelt structure
476 C----------------------------------------------------------------------------
477 C

479  IF (iddlevel == 0) my_allocate(seatbelt_tab,n_seatbelt)
480  my_allocate(tag_mat_2d,nummat)
481  tag_mat_2d(1:nummat) = 0
482  IF (nb_2d_seatbelt > 0) THEN
483  my_allocate(tag_shell,numelc)
484  my_allocate(section_mat,nummat)
485  tag_shell(1:numelc) = 0
486  section_mat(1:nummat) = zero
487  ENDIF
488 C
489  DO i=1,n_seatbelt
490  compt = 0
491  compt_2d = 0
492  seatbelt_tab(i)%NFRAM = 1
493  seatbelt_tab(i)%NNOD = 0
494  seatbelt_tab(i)%ELEM_SIZE = zero
495  DO j=1,compt_belt_end
496  IF (tag_fram_seatbelt(j)==i) THEN
497  seatbelt_tab(i)%NNOD = seatbelt_tab(i)%NNOD + nnod_fram_seatbelt(j)
498  seatbelt_tab(i)%NFRAM = belt_end_nfram(j)
499  seatbelt_tab(i)%SECTION = belt_end_section(j)
500  ENDIF
501  ENDDO
502  DO j=1,numelr
503  IF (tag_res(j) == i) THEN
504 C-- count of 1d elements of the seatbelt
505  compt = compt + 1
506  mid = ixr(5,j)
507  IF (tag_mat_2d(mid)==0) tag_mat_2d(mid) = -mid
508 C-- count and tag of 2d elements of the seatbelt
509  DO k=2,3

512  elem_cur = nod2elc(l)
513  mid_2d = ixc(1,elem_cur)
514  mtyp = ipm(2,mid_2d)
515  IF (mtyp==119) THEN
516  IF (tag_shell(elem_cur)==0) THEN
517  tag_shell(elem_cur) = i
518  compt_2d = compt_2d + 1
519  tag_mat_2d(mid) = mid_2d

521  section_mat(mid_2d) = seatbelt_tab(i)%SECTION
522  ELSEIF (abs(seatbelt_tab(i)%SECTION-section_mat(mid_2d)) > em05) THEN
523  CALL ancmsg(msgid=2075,
524  . msgtype=msgerror,
